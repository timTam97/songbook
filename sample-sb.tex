\begin{song}{Dear Savior, Thou art mine}{D}
  {}
  {}
  {}
  {}

  \renewcommand{\RevDate}{February~11,~1993}
  \SBRef{Hosanna! Music Book~I}{\#65}

  \begin{SBVerse}
    Dear Savior, Thou art mine,

    How sweet the thought to me;

    Let me repeat Thy name,

    And lift my heart to Thee.
  \end{SBVerse}

  \begin{SBOpGroup}
    Mine! Mine! Mine!
I know Thou art mine;
Savior, dear Savior,
  I know Thou art mine.

  \end{SBOpGroup}

  \begin{SBVerse}
    Thou art the sinner’s friend,
So I Thy friendship claim,
A sinner saved by grace,
When Thy sweet message came.
  \end{SBVerse}

  \begin{SBVerse}
    My hardened heart was touched;
Thy pard’ning voice I heard;
And joy and peace came in
While list’ning to Thy word.
  \end{SBVerse}

  \begin{SBVerse}
    
So let me sing Thy praise,
So let me call Thee mine.
I cannot doubt Thy word,
I know that I am Thine.
  \end{SBVerse}
\end{song}


\begin{song}{A Mind At Perfect Peace With God}{G}
  {\SBPubDom}
  {Horatius Bonar}
  {}
  {}

  \begin{SBVerse}
    A mind at perfect peace with God;

    O what a word is this!

    A sinner reconciled through blood;

    This, this indeed is peace.
  \end{SBVerse}

  \begin{SBVerse}
    By nature and by practice far,

    How very far from God;

    Yet now by grace brought nigh to Him,

    Through faith in Jesus' blood.
  \end{SBVerse}

  \begin{SBVerse}
    So nigh, so very nigh to God,

    I cannot nearer be;

    For in the person of His Son

    I am as near as He.
  \end{SBVerse}

  \begin{SBVerse}
    So dear, so very dear to God,

    More dear I cannot be;

    The love wherewith He loves the Son,

    Such is His love to me.
  \end{SBVerse}

  \begin{SBVerse}
    Why should I ever anxious be,

    Since such a God is mine?

    He watches o'er me night and day,

    And tells me ``Mine is thine.''
  \end{SBVerse}
\end{song}


\begin{song}{He Hideth My Soul}{}

  {}

  \begin{SBVerse}

    A wonderful Savior to me;

    He hideth my soul in the cleft of the rock,

    Where rivers of pleasure I see.
  \end{SBVerse}

    He hideth my soul in the cleft of the rock

    He hideth my life in the depths of His love,

  \end{SBOpGroup}

  \begin{SBVerse}
    A wonderful Savior is Jesus my Lord,

    He taketh my burden away;

    He holdeth me up and I shall not be moved,

    He giveth me strength as my day.
  \end{SBVerse}

  \begin{SBVerse}
    With numberless blessings each moment He crowns,

    And, filled with His fulness divine,

    I sing in my rapture, oh, glory to God,

    For such a Redeemer as mine!
  \end{SBVerse}

  \begin{SBVerse}
    When clothed in His brightness, transported I rise

    To meet Him in clouds of the sky,

    His perfect salvation, His wonderful love,

    I'll shout with the millions on high.
  \end{SBVerse}
\end{song}


\begin{song}{Redemption Ground}{}
  {\SBPubDom}
  {}
  {}
  {}

  \begin{SBVerse}
    Come sing, my soul, and praise the Lord,

    Who hath redeemed thee by His blood;

    Delivered thee from chains that bound,

    And brought thee to redemption ground.
  \end{SBVerse}

  \begin{SBOpGroup}
    Redemption ground, the ground of peace!

    Redemption ground, O wondrous grace!

    Here let our praise to God abound!

    Who saves us on redemption ground.
  \end{SBOpGroup}

  \begin{SBVerse}
    Once from my God I wandered far,

    And with His holy will made war;

    But now my songs to God abound;

    I'm standing on redemption ground.
  \end{SBVerse}

  \begin{SBVerse}
    O joyous hour! when God to me

    A vision gave of Calvary;

    My bonds were loos'd---my soul unbound,

    I sang upon redemption ground.
  \end{SBVerse}

  \begin{SBVerse}
    No works of merit now I plead,

    But Jesus take, for all my need;

    No righteousness in me is found,

    Except upon redemption ground.
  \end{SBVerse}

  \begin{SBVerse}
    Come, weary soul, and here find rest;

    Accept redemption, and be blest:

    The Christ who died, by God is crowned

    To pardon on redemption ground.
  \end{SBVerse}
\end{song}


\begin{song}{Day By Day}{}
  {\SBPubDom}
  {Lina Sandell}
  {}
  {}

  \begin{SBVerse}
    Day by day, and with each passing moment,

    Strength I find to meet my trials here;

    Trusting in my Father's wise bestowment,

    I've no cause for worry or for fear.

    He, whose heart is kind beyond all measure,

    Gives unto each day what He deems best,

    Lovingly its part of pain and pleasure,

    Mingling toil with peace and rest.
  \end{SBVerse}

  \begin{SBVerse}
    Every day the Lord Himself is near me,

    With a special mercy for each hour;

    All my cares He fain would bear and cheer me,

    He whose name is Counsellor and Pow'r.

    The protection of His child and treasure

    Is a charge that on Himself He laid;

    ``As thy days, thy strength shall be in measure,''

    This the pledge to me He made.
  \end{SBVerse}

  \begin{SBVerse}
    Help me then, in every tribulation,

    So to trust Thy promises, O Lord,

    That I lose not faith's sweet consolation,

    Offered me within Thy holy Word.

    Help me, Lord, when toil and trouble meeting,

    E'er to take, as from a father's hand,

    One by one, the days, the moments fleeting,

    Till with Christ the Lord I stand.
  \end{SBVerse}
\end{song}
